let allWordUnits = {}, currentWord = "", currentSpelling = [], currentWordList = [], wordIndex = 0, errorList = []; let isGameActive = false, isReviewMode = false, flatUnits = []; const synth = window.speechSynthesis; const praiseWords = ["Great job", "Excellent", "Amazing", "Well done"]; const azLayout = ["ABCDEFGHI", "JKLMNOPQR", "STUVWXYZ"]; // 初始化選單：簡化迴圈寫法 function initSelection() { const startSel = document.getElementById('start-unit-select'); const endSel = document.getElementById('end-unit-select'); Object.entries(allWordUnits).forEach(([g, units]) => { Object.keys(units).forEach(u => { const idx = flatUnits.length; flatUnits.push({ grade: g, unit: u }); azLayout.forEach((row, idx) => { const div = document.createElement('div'); div.className = 'kb-row'; row.split('').forEach(l => { const b = document.createElement('button'); b.className = 'key'; b.textContent = l; b.onclick = () => handleKeyClick(l); div.appendChild(b); }); if (idx === 2) { const del = document.createElement('button'); del.id = 'delete-key'; del.className = 'key'; del.textContent = 'DEL'; del.onclick = handleDelete; div.appendChild(del); } kb.appendChild(div); }); } function handleNavAction() { if (!isGameActive) { startGame(); document.getElementById('nav-btn').textContent = '➡️ 下一個單字'; document.getElementById('repeat-btn').disabled = false; currentWordList = flatUnits.slice(s, e + 1).flatMap(item => allWordUnits[item.grade][item.unit]); currentWordList.sort(() => Math.random() - 0.5); wordIndex = 0; errorList = []; isGameActive = true; isReviewMode = false; loadNextWord(); } function handleMainAction() { const btn = document.getElementById('check-btn'); if (btn.textContent === '✅ 對答案') checkSpelling(); else if (btn.textContent === '🔄 重試') { resetInput(); btn.textContent = '✅ 對答案'; btn.disabled = true; } } function loadNextWord() { if (wordIndex >= currentWordList.length) { isReviewMode ? endGame() : handleRoundEnd(); return; } const data = currentWordList[wordIndex]; document.getElementById('check-btn').textContent = '✅ 對答案'; document.getElementById('check-btn').disabled = true; document.getElementById('hint-display').textContent = `[${isReviewMode?'複習':(wordIndex+1)+'/'+currentWordList.length}] ${data.hint}`; let lIdx = 0; currentWord.split('').forEach(char => { const div = document.createElement('div'); if (char === ' ') { div.className = 'space-slot'; } else { div.className = 'letter-slot'; div.id = `slot-${lIdx++}`; const slot = document.getElementById(`slot-${nextIdx}`); slot.textContent = l.toLowerCase(); slot.classList.remove('slot-wrong'); document.getElementById('check-btn').disabled = false; } } function handleDelete() { for (let i = currentSpelling.length - 1; i >= 0; i--) { const slot = document.getElementById(`slot-${i}`); if (slot.textContent && !slot.classList.contains('slot-correct')) { slot.textContent = ''; slot.classList.remove('slot-wrong'); currentSpelling[i] = ""; break; } } document.getElementById('check-btn').disabled = !currentSpelling.some(c => c !== ""); } function checkSpelling() { const target = currentWord.replace(/\s/g, '').toLowerCase(); let allCorrect = true; currentSpelling.forEach((char, i) => { const slot = document.getElementById(`slot-${i}`); if (char === target[i]) { slot.className = 'letter-slot slot-correct'; } else { slot.className = 'letter-slot slot-wrong'; allCorrect = false; } }); if (allCorrect) { document.getElementById('feedback').textContent = '🌟 太棒了！'; if (isReviewMode) removeWrongWord(currentWord); readWord(praiseWords[Math.floor(Math.random() * praiseWords.length)]); setTimeout(() => { wordIndex++; loadNextWord(); }, 1200); } else { document.getElementById('feedback').textContent = '💡 再試一次！'; saveWrongWord(currentWordList[wordIndex]); if (!errorList.includes(currentWordList[wordIndex])) errorList.push(currentWordList[wordIndex]); readWord("Try again"); const btn = document.getElementById('check-btn'); btn.textContent = '🔄 重試'; } } function resetInput() { const target = currentWord.replace(/\s/g, '').toLowerCase(); currentSpelling.forEach((char, i) => { if (char !== target[i]) { currentSpelling[i] = ""; const slot = document.getElementById(`slot-${i}`); slot.textContent = ''; slot.className = 'letter-slot'; } }); } function handleSkip() { const target = currentWord.replace(/\s/g, '').toLowerCase(); currentSpelling.forEach((char, i) => { const slot = document.getElementById(`slot-${i}`); slot.className = (char === target[i]) ? 'letter-slot slot-correct' : 'letter-slot slot-wrong'; slot.textContent = target[i]; }); document.getElementById('feedback').textContent = `📝 已記錄在待複習清單裡`; saveWrongWord(currentWordList[wordIndex]); if (!errorList.includes(currentWordList[wordIndex])) errorList.push(currentWordList[wordIndex]); synth.speak(utter); } function getStoredWords() { return JSON.parse(localStorage.getItem('wrongWords') || '[]'); } function updateWrongWordCount() { document.getElementById('review-book-btn').textContent = `📝 錯題本 (${getStoredWords().length})`; wordIndex = 0; errorList = []; isReviewMode = true; loadNextWord(); } else endGame(); } function endGame() { isGameActive = false; document.getElementById('hint-display').textContent = '🎊 恭喜妳完成練習！'; document.getElementById('nav-btn').textContent = '🚀 開始遊戲'; ['repeat-btn', 'check-btn'].forEach(id => document.getElementById(id).disabled = true); } function startWrongWordMode() { const b = getStoredWords(); if (!b.length) return alert("沒有錯題！"); currentWordList = b.sort(() => Math.random() - 0.5); wordIndex = 0; errorList = []; isGameActive = true; isReviewMode = true; document.getElementById('nav-btn').textContent = '➡️ 下一個單字'; document.getElementById('repeat-btn').disabled = false; loadNextWord(); } window.onload = async () => { synth.getVoices(); if (speechSynthesis.onvoiceschanged !== undefined) speechSynthesis.onvoiceschanged = () => synth.getVoices(); try { const r = await fetch('https://raw.githubusercontent.com/liliancir0125/spell-game/refs/heads/main/word_bank.json'); allWordUnits = await r.json(); initSelection(); createKeyboard(); } catch (e) { document.getElementById('hint-display').textContent = "❌ 無法載入單字庫"; } };
